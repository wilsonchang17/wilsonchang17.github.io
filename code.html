1.Two sum

C++
Diffficulty: easy

            
class Solution {
    public:
        vector<int> twoSum(vector<int>& nums, int target) {
            vector<int> arr;
            for(int i=0; i<nums.size(); i++){
                for(int j=i+1; j<nums.size(); j++){
                    if(nums[i]+nums[j]==target){
                        arr.push_back(i);
                        arr.push_back(j);
                        break;
                    }
                }
            }
            return (arr);             
        }
};

        {
            x1/=10;
            count++;
        }

21. Merge Two Sorted Lists

Python
Diffficulty: easy

            
# Definition for singly-linked list.
# class ListNode(object):
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution(object):
    def mergeTwoLists(self, l1, l2):
        dummy = cur = ListNode(0)
        while(l1!=None and l2 != None):
            if(l1.val<l2.val):
                cur.next = l1
                l1 = l1.next
            else:
                cur.next = l2
                l2 = l2.next
            cur = cur.next
        cur.next = l1 or l2
        return dummy.next

27. Remove Element

Diffficulty: easy

            
class Solution
{
public:
    int removeElement(vector<int> &nums, int val)
    {
        int count = 0;
        if(nums.size()==0){
            return 0;
        }
        for (int i = 0; i < nums.size(); i++){
            if (nums[i] == val){
               nums[i] = -1;
               count++;
            }
        }
        for (int i = 0; i < nums.size(); i++){
            for(int j= nums.size()-1;j>i; j--){
                if(nums[i]==-1){
                    if(nums[j]!=-1){
                        swap(nums[i],nums[j]);
                        break;
                    }
                }
            }
        }
        return nums.size()-count;
    }
};

            return 0;
        }

            return 0;
        }
        int mark = 0;
        for(int i=0; i<haystack.size(); i++){
            if(needle[temp]==haystack[i]){
                temp++;
                count++;

                    mark = i;
                    break;
                }
            }
            else{
                i = i-count;
                temp=0;
                count = 0;
            }

            return mark-needle.size()+1;
        }

            return -1;
        }

class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        int l = 0, r = nums.size()-1;
        int mid;
        while(l<=r){
            mid = l+(r-l)/2;
            if(nums[mid] < target){
                l = mid+1;
            }
            else{
                r = mid-1;
            }
        }
        return l;
    }

                temp.push_back(1);
                temp.push_back(0);
                return temp;
            }

                return digits;
            }
            
        }

        for(int i=digits.size()-1; i>0; i--){

                digits[i-1] +=1;
                digits[i] = 0;

# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def deleteDuplicates(self, head):
        cur = head
        while(cur!=None):
            if(cur.next!=None and cur.next.val==cur.val):
                cur.next = cur.next.next
            else:
                cur = cur.next
        return head

94. Binary Tree Inorder Traversal

Python
Diffficulty: easy

            
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def inorderTraversal(self, root: TreeNode) -> List[int]:
        ans = []
        def find(node):
            if (node == None):
                return
            find(node.left)
            ans.append(node.val)
            find(node.right)
        find(root)
        return (ans)

100. Same Tree

Python
Diffficulty: easy

            
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def isSameTree(self, p: TreeNode, q: TreeNode) -> bool:
        def find(p,q):
            if(p==None and q==None):
                return True
            elif((p==None and q!= None) or (p!=None and q==None) or (q.val!=p.val)):
                return False
            return find(p.left,q.left) and find(p.right,q.right)
        return find(p,q)

101. Symmetric Tree

Python
Diffficulty: easy

            
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def isSymmetric(self, root: TreeNode) -> bool:
        def find(r,l):
            if(r == None and l==None):
                return True
            if(r!=None and l!=None and r.val==l.val):
                return find(r.right,l.left) and find(r.left,l.right)
        return find(root.right,root.left)
